FROM maven:3.6.2-ibmjava-8-alpine

RUN mkdir -p ${app-home}
COPY src ${app-home}/src
COPY target ${app-home}/target
COPY pom.xml ${app-home}/pom.xml
WORKDIR ${app-home}

EXPOSE ${port}/tcp

ENV TZ=${timezone}
RUN ln -snf /usr/share/zoneinfo/$TZ /etc/localtime && echo $TZ > /etc/timezone

ENTRYPOINT ["mvn", "spring-boot:run"]
